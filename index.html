<script>

  const levelSelect = document.getElementById('level');

  const tablaSelect = document.getElementById('tabla');


  function showIntro() {

    document.getElementById('intro').classList.remove('hidden');

    document.getElementById('level1').classList.add('hidden');

    document.getElementById('level2').classList.add('hidden');

    document.getElementById('level3').classList.add('hidden');

  }


  function setupMultiplication() {

    const level = levelSelect.value;

    const tabla = parseInt(tablaSelect.value);


    document.getElementById('intro').classList.add('hidden');


    if(level === '1'){

      showLevel1(tabla);

    } else if(level === '2'){

      showLevel2(tabla);

    } else if(level === '3'){

      showLevel3(tabla);

    }

  }


  function showLevel1(tabla) {

    const container = document.getElementById('level1-content');

    container.innerHTML = '';

    for(let i=0; i<=10; i++){

      const p = document.createElement('p');

      p.textContent = `${tabla} × ${i} = ${tabla*i}`;

      container.appendChild(p);

    }

    document.getElementById('level1').classList.remove('hidden');

  }


  function showLevel2(tabla) {

    const container = document.getElementById('level2-content');

    container.innerHTML = '';

    for(let i=0; i<=10; i++){

      const div = document.createElement('div');

      const label = document.createElement('label');

      label.textContent = `${tabla} × ${i} = `;

      const input = document.createElement('input');

      input.type = 'number';

      input.min = 0;

      input.max = 100;

      input.dataset.answer = tabla*i;

      input.addEventListener('input', function(){

        if(parseInt(this.value) === tabla*i){

          this.style.backgroundColor = '#b9f6ca'; // verde claro

        } else {

          this.style.backgroundColor = '#ffcdd2'; // rojo claro

        }

      });

      div.appendChild(label);

      div.appendChild(input);

      container.appendChild(div);

    }

    document.getElementById('level2').classList.remove('hidden');

  }


  function showLevel3(tabla) {

    const container = document.getElementById('level3-content');

    container.innerHTML = '';

    for(let i=0; i<=10; i++){

      const div = document.createElement('div');

      const label = document.createElement('label');

      label.textContent = `${tabla} × ${i} = `;

      const input = document.createElement('input');

      input.type = 'number';

      input.min = 0;

      input.max = 100;

      input.dataset.answer = tabla*i;

      div.appendChild(label);

      div.appendChild(input);

      container.appendChild(div);

    }

    const checkButton = document.createElement('button');

    checkButton.textContent = 'Comprobar respuestas';

    checkButton.onclick = checkAnswers;

    container.appendChild(checkButton);

    document.getElementById('level3').classList.remove('hidden');

    document.getElementById('feedback').textContent = '';

  }


  function checkAnswers() {

    const inputs = document.querySelectorAll('#level3-content input');

    let corrects = 0;

    inputs.forEach(input => {

      const answer = parseInt(input.dataset.answer);

      if(parseInt(input.value) === answer){

        input.style.backgroundColor = '#b9f6ca';

        corrects++;

      } else {

        input.style.backgroundColor = '#ffcdd2';

      }

    });

    const feedback = document.getElementById('feedback');

    feedback.textContent = `Has acertado ${corrects} de ${inputs.length} preguntas.`;

  }


  // Mostrar inicio al cargar

  showIntro();
